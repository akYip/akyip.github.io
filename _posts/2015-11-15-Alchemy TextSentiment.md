Alchemy TextSentiment API analyzes text and tells me it is positive or negative.

<span style="font-size:10px;"><a href="http://www.igosso.net/flk/991439100.html" target="_blank"><img src="https://farm2.staticflickr.com/1368/991439100_16f39bfaa6_m.jpg" alt="" /></a><br />Trump's Birdnest / AZRainman</span>


~~~~
$APPKEY="YOUR APP KEY";
$text='"I\'m more disappointed in many ways with the Republicans," Trump said.
"They have this great indignation, whether it\'s Benghazi or the emails... nothing ever happens."';//put your text here


$url = "http://access.alchemyapi.com/calls/text/TextGetTextSentiment".
"?text=".urlencode($text)."&amp;outputMode=json&amp;apikey=".$APPKEY;

$o = array(
'Content-Type: application/x-www-form-urlencoded'
);

$curl = curl_init();
curl_setopt($curl, CURLOPT_POST, true);
curl_setopt($curl, CURLOPT_HTTPHEADER, $o);
curl_setopt($curl, CURLOPT_URL, $url);
curl_setopt($curl, CURLOPT_RETURNTRANSFER, true);
curl_setopt($curl, CURLOPT_SAFE_UPLOAD, true);
curl_setopt($curl, CURLOPT_SSL_VERIFYPEER, false);
$result = curl_exec($curl);
curl_close($curl);
print $result;
~~~~
